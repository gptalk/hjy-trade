def init_db():
    conn = sqlite3.connect('data/stock_data.db')
    cursor = conn.cursor()

    cursor.execute('''
        CREATE TABLE IF NOT EXISTS stocks (
            code TEXT PRIMARY KEY,
            name TEXT NOT NULL,
            market TEXT DEFAULT 'A股'
        )
    ''')

    cursor.execute('''
        CREATE TABLE IF NOT EXISTS klines (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            code TEXT,
            date TEXT,
            open REAL,
            high REAL,
            low REAL,
            close REAL,
            volume INTEGER,
            UNIQUE(code, date)
        )
    ''')

    cursor.execute('''
        CREATE TABLE IF NOT EXISTS strategies (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT UNIQUE,
            description TEXT,
            conditions TEXT,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
    ''')

    cursor.execute('''
        CREATE TABLE IF NOT EXISTS backtest_records (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            strategy_id INTEGER,
            stock_code TEXT,
            start_date TEXT,
            end_date TEXT,
            initial_capital REAL,
            final_capital REAL,
            return_rate REAL,
            win_rate REAL,
            max_drawdown REAL,
            sharpe_ratio REAL,
            trade_count INTEGER,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
    ''')

    cursor.execute('''
        CREATE TABLE IF NOT EXISTS watchlists (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            group_name TEXT,
            stock_code TEXT,
            added_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
    ''')

    conn.commit()
    conn.close()
```

    df['MACD_hist'] = macd['MACDh_12_26_9']

import React, { useEffect, useRef } from 'react';
import * as echarts from 'echarts';

const KLineChart = ({ data, indicators = [] }) => {
  const chartRef = useRef(null);

  useEffect(() => {
    if (!chartRef.current || !data.length) return;

    const chart = echarts.init(chartRef.current);
    const dates = data.map(d => d.date);
    const klineData = data.map(d => [d.open, d.close, d.low, d.high]);
    const volumes = data.map(d => d.volume);

    const option = {
      tooltip: { trigger: 'axis', axisPointer: { type: 'cross' } },
      grid: [
        { left: '10%', right: '8%', top: '10%', height: '50%' },
        { left: '10%', right: '8%', top: '65%', height: '20%' }
      ],
      xAxis: [{ type: 'category', data: dates }, { type: 'category', gridIndex: 1, data: dates }],
      yAxis: [{ scale: true }, { scale: true, gridIndex: 1 }],
      series: [
        { name: 'KLine', type: 'candlestick', data: klineData },
        { name: 'Volume', type: 'bar', xAxisIndex: 1, yAxisIndex: 1, data: volumes }
      ]
    };

    chart.setOption(option);
    return () => chart.dispose();
  }, [data, indicators]);

  return <div ref={chartRef} style={{ width: '100%', height: '500px' }} />;
};

export default KLineChart;
```

- [ ] **Step 2: 创建回测引擎**

```python
class BacktestEngine:
    def __init__(self, initial_capital=10000):
        self.initial_capital = initial_capital
        self.positions = []
        self.trades = []
        self.capital = initial_capital

    def run(self, klines, strategy):
        """执行回测"""
        for i, kline in enumerate(klines):
            signal = self.check_signal(kline, strategy, klines[:i])
            if signal == 'BUY':
                self.buy(kline)
            elif signal == 'SELL':
                self.sell(kline)

        return self.generate_report()

    def check_signal(self, kline, strategy, history):
        """检查交易信号"""
        # 简单示例: MA金叉
        if len(history) < 20:
            return None

        ma5 = sum([k['close'] for k in history[-5:]]) / 5
        ma20 = sum([k['close'] for k in history[-20:]]) / 20
        prev_ma5 = sum([k['close'] for k in history[-6:-1]]) / 5
        prev_ma20 = sum([k['close'] for k in history[-21:-1]]) / 20

        if prev_ma5 <= prev_ma20 and ma5 > ma20:
            return 'BUY'
        elif prev_ma5 >= prev_ma20 and ma5 < ma20:
            return 'SELL'
        return None

    def buy(self, kline):
        shares = int(self.capital / kline['close'] / 100) * 100
        if shares > 0:
            self.positions.append({'price': kline['close'], 'shares': shares, 'date': kline['date']})
            self.capital -= shares * kline['close']

    def sell(self, kline):
        for pos in self.positions[:]:
            self.trades.append({
                'buy_date': pos['date'], 'buy_price': pos['price'],
                'sell_date': kline['date'], 'sell_price': kline['close'],
                'shares': pos['shares']
            })
            self.capital += pos['shares'] * kline['close']
            self.positions.remove(pos)

    def generate_report(self):
        if not self.trades:
            return {'return_rate': 0, 'win_rate': 0, 'trade_count': 0}
        # 计算收益率、胜率等
        returns = [(t['sell_price'] - t['buy_price']) / t['buy_price'] for t in self.trades]
        wins = sum(1 for r in returns if r > 0)
        return {
            'final_capital': self.capital + sum(p['shares'] * p['price'] for p in self.positions),
            'return_rate': (self.capital - self.initial_capital) / self.initial_capital * 100,
            'win_rate': wins / len(returns) * 100 if returns else 0,
            'trade_count': len(self.trades)
        }
```

@bp.route('/analyze', methods=['POST'])
def analyze_stock():
    data = request.json
    code = data.get('code')
    klines = get_klines_from_db(code)

    if len(klines) < 30:
        return jsonify({'error': '数据不足'})

    # 简单技术分析
    latest = klines[-1]
    ma5 = sum([k['close'] for k in klines[-5:]]) / 5
    ma20 = sum([k['close'] for k in klines[-20:]]) / 20

    analysis = {
        'code': code,
        'price': latest['close'],
        'trend': '上涨' if ma5 > ma20 else '下跌',
        'ma5': ma5,
        'ma20': ma20,
        'volume_ratio': latest['volume'] / (sum([k['volume'] for k in klines[-20:-1]]) / 20),
        'signal': '中性',
        'support': min([k['low'] for k in klines[-10:]]),
        'resistance': max([k['high'] for k in klines[-10:]])
    }

    if ma5 > ma20 and latest['volume'] > klines[-2]['volume']:
        analysis['signal'] = '买入信号'
    elif ma5 < ma20:
        analysis['signal'] = '观望'

    return jsonify(analysis)
